class Solution {
    public int minCostClimbingStairs(int[] cost) {
    //确定楼梯的终点
    int n =cost.length;
    //dp[i]=到达第i个台阶的最小花费
    int[] dp=new int[n];
    //初始化刚开始的 没有初始化最刚开始那么后面的台阶就比不了
    dp[0]=cost[0];
    dp[1]=cost[1];

    //后面主动碰头 选最少的
    for(int i=2; i<n;i++){
        int fromPrev1=dp[i-1];
        int fromPrev2=dp[i-2];
        dp[i] = Math.min(fromPrev1, fromPrev2) + cost[i];
    }
     // 最后一步到楼顶：可以站在 n-1 或 n-2
        return Math.min(dp[n - 1], dp[n - 2]);
    }
}
