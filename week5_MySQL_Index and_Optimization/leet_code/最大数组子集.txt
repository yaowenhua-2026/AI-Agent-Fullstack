class Solution {
    public int maxSubArray(int[] nums) {
        int prev = nums[0];   // 以第 0 天结尾的最大子数组和
        int max = nums[0];    // 到目前为止的全局最大

        for (int i = 1; i < nums.length; i++) {
            int curr = Math.max(nums[i], prev + nums[i]);
            max = Math.max(max, curr);
            prev = curr;
        }
        return max;
    }
}
