class Solution {
    public int lengthOfLongestSubstring(String s) {
//Character,Integer
    Map<Character,Integer> map=new HashMap<>();
    int left=0;
    int maxLen=0;
    int right=s.length();

    for(int i=0;i<right;i++){
        //获取第i为的值
            char c=s.charAt(i);
        //拿到值进行比对
        if(map.containsKey(c)){
            //如果重复了 那么就往前走
            left=Math.max(left,map.get(c)+1);
        }
        //继续添加进map中
        map.put(c,i);
        //记录最大子串值
        maxLen=Math.max(maxLen,i-left+1);

    }
    return maxLen;
}
}
class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> res = new ArrayList<>();

        for (int i = 0; i < nums.length - 2; i++) {
            // 去重 i
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            int left = i + 1, right = nums.length - 1;

            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];

                if (sum == 0) {
                    // ✅ 正确：添加完整的三元组
                    res.add(Arrays.asList(nums[i], nums[left], nums[right]));

                    // 去重 left
                    while (left < right && nums[left] == nums[left + 1]) left++;
                    // 去重 right
                    while (left < right && nums[right] == nums[right - 1]) right--;

                    // 指针移动
                    left++;
                    right--;
                } else if (sum < 0) {
                    left++;
                } else {
                    right--;
                }
            }
        }
        return res;
    }
}
class Solution {
    public int search(int[] nums, int target) {
        if (nums == null || nums.length == 0) return -1;
        
        int left = 0, right = nums.length - 1;
        
        // 第一步：找旋转点（最小值的索引）
        int rotateIndex = 0;
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] < nums[i - 1]) {
                rotateIndex = i;
                break;
            }
        }
        
        // 第二步：确定 target 在哪一段
        if (target >= nums[0] && rotateIndex > 0 && target <= nums[rotateIndex - 1]) {
            // target 在左段（大数区）
            left = 0;
            right = rotateIndex - 1;
            //这里还判断是是否有旋转 如果 没有旋转 那么肯定 不能大于最后一个值
        } else if (rotateIndex < nums.length && target >= nums[rotateIndex] && target <= nums[nums.length - 1]) {
            // target 在右段（小数区）
            left = rotateIndex;
            right = nums.length - 1;
        } else {
            return -1;  // target 不在任何一段
        }
        
        // 第三步：二分查找
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                return mid;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        
        return -1;
    }
} 
class Solution {
    public int[][] merge(int[][] intervals) {
        //首先先排序
        Arrays.sort(intervals,(a,b)->Integer.compare(a[0],b[0]));
        //收集结果 
        List<int[]> result=new ArrayList<>();
        int [] current=intervals[0];//母合成区域,第一个区域

        for(int i =1;i<intervals.length;i++){
            if(intervals[i][0]<=current[1]){
                //有重叠：更新终点
                current[1]=Math.max(current[1],intervals[i][1]);

            }else{
                //无重叠把当前区间加入结果 开始新区间
                result.add(current);
                current=intervals[i];
            }

        }
        result.add(current); //最后一个
        return result.toArray(new int[result.size()][]);
    }
}
class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        // 1. 创建最终要返回的二维列表
        List<List<Integer>> result = new ArrayList<>();
        
        // 特殊情况处理：如果根节点是空的，直接返回空列表
        if (root == null) return result;
        
        // 2. 初始化队列：Java 中通常用 LinkedList 来实现 Queue 接口
        Queue<TreeNode> queue = new LinkedList<>();
        // 将根节点放入队列中，作为遍历的起点
        queue.offer(root);
        
        // 3. 开始大循环：只要队列里还有节点，说明还没遍历完
        while (!queue.isEmpty()) {
            // 核心关键点：获取当前这一层有多少个节点
            int levelSize = queue.size();
            // 创建一个小列表，用来存放当前层的所有节点
            List<Integer> currentLevel = new ArrayList<>();
            
            // 4. 局部循环：严格执行 levelSize 次，确保只处理当前层节点
            for (int i = 0; i < levelSize; i++) {
                TreeNode node = queue.poll();
                currentLevel.add(node.val);  // 记录当前节点值
                
                // 左右子节点入队（非空才入队）
                if (node.left != null) queue.offer(node.left);
                if (node.right != null) queue.offer(node.right);
            }
            
            // 5. 当前层处理完成，加入结果
            result.add(currentLevel);
        }
        
        return result;
    }
}
